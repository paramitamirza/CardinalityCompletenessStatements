NOVEL
---
- novel-a.csv

SELECT ?mainObject (COUNT(DISTINCT(?part)) AS ?countPart)
WHERE {
    ?mainObject wdt:P31 wd:Q1667921 .
    ?part wdt:P179 ?mainObject
} GROUP BY ?mainObject

- novel-b.csv

SELECT ?mainObject (COUNT(DISTINCT(?part)) AS ?countPart)
WHERE {
    ?mainObject wdt:P31 wd:Q1667921 .
    ?mainObject wdt:P527 ?part
} GROUP BY ?mainObject
